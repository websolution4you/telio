const { createClient } = require("@supabase/supabase-js");
const fs = require('fs');
const env = fs.readFileSync('.env.local', 'utf8');

function getVal(key) {
    const line = env.split('\n').find(l => l.startsWith(key + '='));
    return line ? line.split('=')[1].trim() : '';
}

async function enableRealtime(url, key, tables) {
    const supabase = createClient(url, key);
    console.log(`--- Enabling Realtime for ${url} ---`);
    for (const table of tables) {
        console.log(`Enabling for ${table}...`);
        // This is a common way to enable realtime via SQL if using service role
        const { error } = await supabase.rpc('exec_sql', { 
            sql: `alter publication supabase_realtime add table ${table};` 
        });
        if (error) {
            // Fallback: try raw query if rpc fails (most likely it will if exec_sql isn't defined)
            console.log(`RPC failed for ${table}, might need manual setup:`, error.message);
        } else {
            console.log(`Enabled ${table} successfully via RPC.`);
        }
    }
}
